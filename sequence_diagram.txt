sequenceDiagram
    actor User
    participant Conductor as Conductor Server
    participant Prepare as Prepare Fork Inputs<br/>(INLINE)
    participant Fork as FORK_JOIN_DYNAMIC
    participant CertHound as cert_hound<br/>(Sub-Workflow)
    participant SSLWorker as check_ssl_certs<br/>(Python Worker)
    participant Join as JOIN
    participant Severity as Compute Severity<br/>(INLINE)
    participant Switch as SWITCH<br/>(Tiered Alert)

    User->>Conductor: POST /workflow/cert_hound_monitor<br/>{clients, expiration_window_days}
    activate Conductor

    Conductor->>Prepare: Evaluate JS expression
    activate Prepare
    Prepare-->>Conductor: fork_inputs[] (one per client)
    deactivate Prepare

    Conductor->>Fork: Fan out sub-workflows (optional: true)
    activate Fork

    loop Each client (parallel)
        Fork->>CertHound: Start cert_hound(urls)
        activate CertHound
        CertHound->>SSLWorker: Poll & execute check_ssl_certs
        activate SSLWorker
        SSLWorker-->>CertHound: {expiring_urls, details, totals}
        deactivate SSLWorker
        deactivate CertHound
    end

    Fork->>Join: All sub-workflows complete
    deactivate Fork
    activate Join

    Join-->>Conductor: Merged results (keyed by client)
    deactivate Join

    Conductor->>Severity: Scan results for errors/expiry
    activate Severity
    Severity-->>Conductor: worst_severity
    deactivate Severity

    Conductor->>Switch: Route on severity
    activate Switch

    alt critical
        Switch-->>Conductor: PagerDuty P1 + Email
    else warning
        Switch-->>Conductor: Slack + Email
    else healthy
        Switch-->>Conductor: Slack
    end
    deactivate Switch

    Conductor-->>User: Workflow COMPLETED<br/>{results, worst_severity}
    deactivate Conductor
